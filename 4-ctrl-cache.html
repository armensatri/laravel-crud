public function index() { $search = request('search'); $role = request('role'); $page = request('page', 1); $cacheVersion = Cache::get('users_cache_version', 1); $cachekey = "users_v{$cacheVersion}:search:{$search}:role:{$role}:page:{$page}"; $users = Cache::remember( $cachekey, now()->addMinutes(5), function () { return User::search(request(['search', 'role'])) ->select(['id', 'name', 'username', 'email', 'image', 'role_id', 'url']) ->with(['role']) ->orderBy('id', 'asc') ->paginate(10) ->withQueryString(); } ); return view('backend.manageuser.users.index', [ 'title' => 'Semua data users', 'users' => $users ]); }
